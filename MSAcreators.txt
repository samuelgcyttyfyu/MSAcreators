            const mobileMenu = document.getElementById('mobile-menu');
            
            mobileMenuBtn.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            // Fechar menu mobile ao clicar em um link
            mobileMenu.querySelectorAll('a').forEach(link => {
                link.addEventListener('click', () => {
                    mobileMenu.classList.add('hidden');
                });
            });
            
            // --- Lógica do Modal de Projeto Detalhado ---
            const projectModal = document.getElementById('project-modal');
            const closeProjectModalBtn = document.getElementById('close-project-modal-btn');
            const playerContainer = document.getElementById('video-player-container');
            const portfolioItems = document.querySelectorAll('.portfolio-item');

            const openProjectModal = (element) => {
                const videoType = element.dataset.videoType;
                const videoId = element.dataset.videoId;
                
                if (!videoType || !videoId) return;

                document.getElementById('modal-title').textContent = element.dataset.title;
                document.getElementById('modal-role').textContent = element.dataset.role;
                document.getElementById('modal-description').textContent = element.dataset.description;
                // REMOVIDO: document.getElementById('modal-skills').textContent = element.dataset.skills;
                document.getElementById('modal-software').textContent = element.dataset.software;

                let iframeSrc = '';
                if (videoType === 'vimeo') {
                    iframeSrc = `https://player.vimeo.com/video/${videoId}?autoplay=1&muted=0&color=64B5F6&title=0&byline=0&portrait=0`; // Cor do player vimeo atualizada para Azul Claro
                } else if (videoType === 'youtube') {
                    iframeSrc = `https://www.youtube.com/embed/${videoId}?autoplay=1&rel=0&mute=0`;
                }
                
                // Definindo a classe do iframe para garantir o aspect ratio 9/16 e altura máxima
                playerContainer.innerHTML = `<iframe src="${iframeSrc}" frameborder="0" allow="autoplay; fullscreen; picture-in-picture" allowfullscreen class="h-[85vh] aspect-[9/16]"></iframe>`;
                
                projectModal.classList.remove('hidden');
                projectModal.classList.add('flex');
            };

            const closeProjectModal = () => {
                playerContainer.innerHTML = '';
                projectModal.classList.add('hidden');
                projectModal.classList.remove('flex');
            };

            portfolioItems.forEach(item => {
                item.addEventListener('click', () => {
                    openProjectModal(item);
                });
            });

            closeProjectModalBtn.addEventListener('click', closeProjectModal);
            projectModal.addEventListener('click', (event) => {
                if (event.target === projectModal) closeProjectModal();
            });

            // --- Fechar Modais com a tecla ESC ---
            document.addEventListener('keydown', (event) => {
                if (event.key === 'Escape') {
                    closeProjectModal();
                }
            });

            // --- Lógica do Carrossel de Logos ---
            const scrollers = document.querySelectorAll(".scroller");
            if (scrollers.length > 0) {
                scrollers.forEach((scroller) => {
                    const scrollerInner = scroller.querySelector(".scroller__inner");
                    const scrollerContent = Array.from(scrollerInner.children);

                    scrollerContent.forEach((item) => {
                        const duplicatedItem = item.cloneNode(true);
                        duplicatedItem.setAttribute("aria-hidden", true);
                        scrollerInner.appendChild(duplicatedItem);
                    });
                });
            }

            // --- Efeito de Fade-in nas seções ao rolar ---
            const sections = document.querySelectorAll('section');
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('fade-in-section');
                    }
                });
            }, { threshold: 0.1 });

            sections.forEach(section => {
                observer.observe(section);
            });
        });

        // --- GEMINI AI INTEGRATION (ROTEIRO VIRAL) ---
        const btnGenerate = document.getElementById('btn-generate');
        const aiOutput = document.getElementById('ai-output');
        const aiResultContainer = document.getElementById('ai-result-container');
        const apiKey = ""; // Chave injetada automaticamente pelo ambiente

        btnGenerate.addEventListener('click', async () => {
            const niche = document.getElementById('ai-niche').value;
            const goal = document.getElementById('ai-goal').value;

            if (!niche) {
                alert("Por favor, preencha o Nicho.");
                return;
            }

            // UI de Carregamento
            const originalBtnText = btnGenerate.innerHTML;
            btnGenerate.innerHTML = `<div class="loader"></div> Gerando Roteiro...`;
            btnGenerate.disabled = true;
            aiResultContainer.classList.add('hidden');
            aiOutput.innerHTML = '';

            // Prompt Roteiro Detalhado (Atualizado sem nome do produto)
            const prompt = `Aja como um roteirista de vídeo profissional para mídias sociais (TikTok/Reels).
            Escreva um Roteiro Comercial de 1 minuto para um negócio ou serviço típico do nicho "${niche}" com o objetivo de "${goal}".
            
            Use EXATAMENTE este formato estruturado (sem markdown, use HTML simples como <h3>, <ul>, <li>, <strong>):
            
            <h3>🎬 Roteiro Comercial – Nicho: ${niche} (1min)</h3>
            
            <div style="margin-bottom: 15px;">
                <strong>[0s–5s] – Abertura (Gancho)</strong>
                <ul>
                    <li>📽️ <strong>Imagem:</strong> [Descreva a cena visual]</li>
                    <li>🎵 <strong>Trilha:</strong> [Sugestão de música]</li>
                    <li>🎙️ <strong>Narração:</strong> "[Texto falado]"</li>

            </div>

            <div style="margin-bottom: 15px;">
                <strong>[5s–15s] – Apresentação</strong>
                <ul>
                    <li>📽️ <strong>Imagem:</strong> [Descreva a cena visual]</li>
                    <li>🎙️ <strong>Narração:</strong> "[Texto falado]"</li>
                </ul>
            </div>

            <div style="margin-bottom: 15px;">
                <strong>[15s–30s] – Benefícios/Experiência</strong>

                <strong>[30s–50s] – Prova Social/Resultados</strong>

                </ul>

            try {
                // Implementação de Exponential Backoff para a chamada da API
                let attempt = 0;
                const maxRetries = 5;
                let response;

                while (attempt < maxRetries) {
                    try {
                        response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json'
                            },
                            body: JSON.stringify({

                                    parts: [{
                                        text: prompt
                                    }]
                                }]
                            })
                        });
                        
                        if (!response.ok) {
                            throw new Error(`HTTP error! status: ${response.status}`);
                        }
                        
                        // Sai do loop se a resposta for bem-sucedida
                        break; 
                    } catch (fetchError) {
                        attempt++;
                        if (attempt >= maxRetries) {
                            throw fetchError; // Falha após a última tentativa
                        }
                        const delay = Math.pow(2, attempt) * 1000;
                        await new Promise(resolve => setTimeout(resolve, delay));

                    }

                // Usar um modal customizado ou div para notificar o erro, em vez de alert()
                aiOutput.innerHTML = `<p class="text-red-600 font-bold">❌ Erro ao comunicar com a IA. Por favor, tente novamente.</p>`;

            } finally {

                btnGenerate.disabled = false;
            }
        });
